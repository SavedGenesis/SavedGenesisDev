# Redirect HTTP -> HTTPS
server {
    listen 80;
    listen [::]:80;
    server_name panel.savedgenesis.com;
    return 301 https://$host$request_uri;
}

# Panel over HTTPS
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name panel.savedgenesis.com;

    root /var/www/pterodactyl/public;
    index index.php;

    # Logs (optional paths)
    access_log /var/log/nginx/pterodactyl.app-access.log;
    error_log  /var/log/nginx/pterodactyl.app-error.log error;

    # Larger uploads + longer scripts (optional)
    client_max_body_size 100m;
    client_body_timeout 120s;

    # SSL certs (Certbot will fill these)
    ssl_certificate     /etc/letsencrypt/live/panel.savedgenesis.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/panel.savedgenesis.com/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;

    # Security headers (good defaults)
    add_header X-Content-Type-Options nosniff;
    add_header X-Frame-Options DENY;
    add_header Referrer-Policy same-origin;
    add_header X-XSS-Protection "1; mode=block";

    # If (and only if) you plan to preload HSTS, read hstspreload.org first.
    # add_header Strict-Transport-Security "max-age=15768000; preload" always;

    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    # PHP-FPM 8.3 on Debian 13 (Trixie)
    location ~ \.php$ {
        include fastcgi_params;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        fastcgi_pass unix:/run/php/php8.3-fpm.sock;
        fastcgi_index index.php;

        # Optional: bump buffers/timeouts for heavier pages
        fastcgi_buffer_size 16k;
        fastcgi_buffers 4 16k;
        fastcgi_connect_timeout 300;
        fastcgi_send_timeout 300;
        fastcgi_read_timeout 300;
    }

    # Block hidden files
    location ~ /\.ht {
        deny all;
    }
}
